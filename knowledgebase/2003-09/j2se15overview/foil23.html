  • Enums may be used in switch case statements

    public void test(IndyCarChassis chassis) {
        switch(chassis) {

            case IndyCarChassis.DALLARA:
                // do something
                break;

            case IndyCarChassis.G_FORCE:
                // do something
                break;

            default:
                // deal with it
        }
    }
}
