    public void doSomething(String arg1, String... varArgs) {}

// varArgs.length = 0;
vat.doSomething("Jeff");
      
// varArgs.length = 1;
vat.doSomething("Jeff", "Betsy");

// varArgs.length = 2;
vat.doSomething("Jeff", "Jake", "Zack");

// varArgs is null
vat.doSomething("Jeff", null);

// varArgs.length = 2 with null values in the array
vat.doSomething("Jeff", null, null);

// varArgs.length = 1 with a null value in the array
String myString = null;
vat.doSomething("Jeff", myString);
